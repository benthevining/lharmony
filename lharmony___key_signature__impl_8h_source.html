35constexpr KeySignature::KeySignature (Type typeToUse, bool isSharps, PitchClass pitchClassOfRoot) noexcept
36 : isFlat (! isSharps), type (typeToUse)
37{
38 struct MajorMinorPair final
39 {
40 int numMajorAccidentals, numMinorAccidentals; // cppcheck-suppress unusedStructMember
41 };
42
43 const auto pair = [root = pitchClassOfRoot.getAsInt(), flat = isFlat]() -> MajorMinorPair
44 {
45 if (flat)
46 {
47 switch (root)
48 {
49 case (0) : return { 0, 3 };
50 case (1) : return { 5, -1 };
51 case (2) : return { -1, 1 };
52 case (3) : return { 3, 6 };
53 case (5) : return { 1, 4 };
54 case (6) : return { 6, -1 };
55 case (7) : return { -1, 2 };
56 case (8) : return { 4, 7 };
57 case (9) : return { -1, 0 };
58 case (10) : return { 2, 5 };
59 case (11) : return { 7, -1 };
60 }
61 }
62
63 switch (root)
64 {
65 case (0) : return { 0, -1 };
66 case (1) : return { 7, 4 };
67 case (2) : return { 2, -1 };
68 case (3) : return { -1, 6 };
69 case (4) : return { 4, 1 };
70 case (6) : return { 6, 3 };
71 case (7) : return { 1, -1 };
72 case (8) : return { -1, 5 };
73 case (9) : return { 3, 0 };
74 case (10) : return { -1, 7 };
75 case (11) : return { 5, 2 };
76 }
77 }();
78
79 if (type == Type::Major)
80 numAccidentals = pair.numMajorAccidentals;
81 else
82 numAccidentals = pair.numMinorAccidentals;
83
84 // LIMES_ASSERT (numAccidentals >= 0 && numAccidentals <= 7);
85}
